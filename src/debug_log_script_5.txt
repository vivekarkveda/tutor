class SquaresAndCircles(Scene):

        # Create the plane

            x_range=[-6, 6, 1],
            y_range=[-4, 6, 1],
            axis_config={"include_numbers": True}
        ).scale(0.8)
        self.play(Create(plane))
        self.wait(1)

        # Create squares and circles
        square_a = Square(side_length=2, color=RED).move_to(LEFT + UP)
        circle_a = Circle(color=RED).surround(square_a)
        square_b = Square(side_length=2, color=BLUE).move_to(RIGHT + UP)
        circle_b = Circle(color=BLUE).surround(square_b)
        squares = VGroup(square_a, square_b)
        circles = VGroup(circle_a, circle_b)

        # Label the squares
        label_a = Text('a', font_size=24, color=RED).next_to(square_a, UP)
        label_b = Text('b', font_size=24, color=BLUE).next_to(square_b, UP)

        # Fade in the squares and labels
        self.play(FadeIn(squares), FadeIn(circles), FadeIn(label_a), FadeIn(label_b))
        self.wait(2)

        # Highlight the expression
        expression = MathTex(r"(a^2 - b^2)", font_size=36).to_edge(UP)
        self.play(Write(expression))
        self.wait(2)

        # Show the difference of squares text
        diff_text = Text("Difference of Squares:", font_size=24).next_to(expression, DOWN)
        factorized_expr = MathTex(r"(a + b)(a - b)", font_size=36).next_to(diff_text, DOWN)
        self.play(Write(diff_text), Write(factorized_expr))
        self.wait(3)

        # Fade to black with ending text
        ending_text = Text("Thank you for watching!", font_size=36).scale(1.5).to_edge(DOWN)
        self.play(FadeOut(plane), FadeOut(squares), FadeOut(circles), FadeOut(label_a), FadeOut(label_b))
        self.play(Write(ending_text))
        self.wait(2)
        self.play(FadeOut(expression, diff_text, factorized_expr, ending_text))

# Run the animation
