import numpy as np

class CircleGeometry(Scene):
    def construct(self):
        # Set up title and canvas
        self.title = Text("Circle Geometry: Rings and Key Terms", font_size=48, font_color=GOLD).to_edge(UP)
        self.camera.frame.save_state()

        # Zoom back to full canvas
        self.camera.frame.restore()

        # Create outer and inner circles
        outer_circle = Circle(radius=2, color="#FF0000", fill_opacity=0.8)
        inner_circle = Circle(radius=1.5, color="#00FF00", fill_opacity=0.6)
        outer_circle.shift(LEFT * 2)  # Adjust position

        # Create labels for circles
        outer_label = Text("Outer Circle", font_size=24).next_to(outer_circle, UP)
        inner_label = Text("Inner Circle", font_size=24).next_to(inner_circle, DOWN)

        # Highlight the ring area
        ring_area = AnnularSector(outer_radius=outer_circle.radius, inner_radius=inner_circle.radius, angle=0, fill_opacity=0.4, color=YELLOW)

        # Create key terms text
        key_terms = [
            "Circle", "Center", "Radius", "Diameter", "Chord", "Segment", "Circumference"
        ]
        key_terms_text = VGroup(*[Text(term, font_size=20) for term in key_terms]).arrange(DOWN, aligned_edge=LEFT)
        key_terms_text.to_edge(RIGHT)

        # Animate the scene
        self.add(self.title)
        self.play(FadeIn(outer_circle), FadeIn(inner_circle), FadeIn(outer_label), FadeIn(inner_label))
        self.play(FadeIn(ring_area))
        self.play(Write(key_terms_text))
        self.wait(2)

        # Animate ring expansion and contraction
        for _ in range(2):
            self.play(ring_area.animate.set_width(ring_area.get_width() * 1.2), run_time=1)
            self.play(ring_area.animate.set_width(ring_area.get_width() / 1.2), run_time=1)

        # Fade out formula and triangle (from previous scene)
        self.play(FadeOut(self.get_mobjects()))

        # Voice-over and wait
        self.wait(45)

        # End with thank you message
        thank_you = Text("Thank you for watching!", font_size=36, font_color=GOLD).to_edge(DOWN)
        self.play(FadeIn(thank_you))
        self.wait()
